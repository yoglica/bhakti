1. Master the Metadata API

The App Router introduced a powerful Metadata API that replaces next/head. Here's how to use it effectively:

Basic Metadata Configuration

// app/layout.js
export const metadata = {
  title: {
    default: 'My Blog',
    template: '%s | My Blog'
  },
  description: 'Your site description',
  keywords: ['nextjs', 'seo', 'blog'],
  authors: [{ name: 'Your Name' }],
  openGraph: {
    title: 'My Blog',
    description: 'Your site description',
    url: 'https://yourdomain.com',
    siteName: 'My Blog',
    images: [{
      url: '/og-image.jpg',
      width: 1200,
      height: 630,
      alt: 'My Blog'
    }],
    type: 'website'
  },
  twitter: {
    card: 'summary_large_image',
    title: 'My Blog',
    description: 'Your site description',
    images: ['/og-image.jpg']
  },
  robots: {
    index: true,
    follow: true,
    googleBot: {
      index: true,
      follow: true
    }
  },
  verification: {
    google: 'your-google-verification-code'
  }
}

2. Generate Dynamic Sitemaps

Create app/sitemap.js to automatically generate an up-to-date sitemap: 

// app/sitemap.js
import { getAllPosts } from '@/lib/posts'

export default async function sitemap() {
  const posts = await getAllPosts()
  
  const blogPosts = posts.map((post) => ({
    url: `https://yourdomain.com/${post.slug}`,
    lastModified: new Date(post.date),
    changeFrequency: 'weekly',
    priority: 0.7
  }))
  
  return [
    {
      url: 'https://yourdomain.com',
      lastModified: new Date(),
      changeFrequency: 'daily',
      priority: 1.0
    },
    {
      url: 'https://yourdomain.com/about',
      lastModified: new Date(),
      changeFrequency: 'monthly',
      priority: 0.5
    },
    ...blogPosts
  ]
}

3. Configure robots.txt

Create app/robots.js to control search engine crawling: 

// app/robots.js
export default function robots() {
  return {
    rules: [
      {
        userAgent: '*',
        allow: '/',
        disallow: ['/api/', '/admin/', '/private/']
      }
    ],
    sitemap: 'https://yourdomain.com/sitemap.xml',
    host: 'https://yourdomain.com'
  }
}

4. Implement JSON-LD for Rich Results

JSON-LD helps Google understand your content and display rich snippets: 

// components/JsonLd.js
export default function JsonLd({ post }) {
  const jsonLd = {
    '@context': 'https://schema.org',
    '@type': 'BlogPosting',
    headline: post.title,
    description: post.description,
    image: post.imageUrl,
    datePublished: post.date,
    dateModified: post.date,
    author: {
      '@type': 'Person',
      name: post.author,
      url: 'https://yourdomain.com/about'
    },
    publisher: {
      '@type': 'Organization',
      name: 'Your Blog',
      logo: {
        '@type': 'ImageObject',
        url: 'https://yourdomain.com/logo.png'
      }
    },
    mainEntityOfPage: {
      '@type': 'WebPage',
      '@id': `https://yourdomain.com/${post.slug}`
    }
  }
  
  return (
    <script
      type="application/ld+json"
      dangerouslySetInnerHTML={{ __html: JSON.stringify(jsonLd) }}
    />
  )
}
